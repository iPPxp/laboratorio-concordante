```text
La relación de satisfacción se denota:

`G ⊨ F`

Las firmas permiten expresar propiedades estructurales sin convertirlas automáticamente en invariantes.
```

```text
## Presv

`Presv` es el subconjunto fijo de Firmas Estructurales designadas como preservables por el protocolo de auditoría.
```

```text
# G_I — Grafo de Acoplamiento de Invariantes

`G_I` es un grafo no dirigido cuyos nodos son invariantes en `Inv(G)`.

Dos invariantes `F_a` y `F_b` están acoplados si existe al menos un embedding testigo de cada uno que comparte algún nodo o arista del modelo `G`.

Esta relación no afirma causalidad ni dirección. Sólo afirma riesgo estructural compartido.

```

```text
# R4 — Propagación de ruptura

R4 calcula la zona de riesgo estructural de una ruptura.

Dado un invariante roto `F_k`, se define:

`Break(F_k)`

como la componente conexa de `F_k` en `G_I`.

## Estados relacionados

- Si no hay ruptura: `APROBADO`.
- Si hay ruptura y no hay conflicto con `Activas(G)`: `SUSPENDIDO`.
- Si `Activas(G) ∩ Break(F_k) ≠ ∅`: `INCONSISTENTE`.
```

```text
from typing import FrozenSet, Set, Dict, Iterable


ESTADO_APROBADO = "APROBADO"
ESTADO_SUSPENDIDO = "SUSPENDIDO"
ESTADO_INCONSISTENTE = "INCONSISTENTE"
ESTADO_NO_AUDITABLE = "NO AUDITABLE"


@dataclass(frozen=True)
class Invariante:
    """
    Firma preservable evaluable por el Auditor.

    id:
        Identificador estable de la firma.
    is_computable:
        Indica si la firma es decidible/auditable.
    is_satisfied:
        Indica si G satisface la firma: G ⊨ F.
    embedding_nodes / embedding_edges:
        Soporte estructural usado para construir G_I por acoplamiento.
    """

    id: str
    is_computable: bool = True
    is_satisfied: bool = True
    embedding_nodes: FrozenSet[str] = field(default_factory=frozenset)
    embedding_edges: FrozenSet[str] = field(default_factory=frozenset)


@dataclass(frozen=True)
class AuditoriaSalida:
    estado: str
    componente_ruptura: FrozenSet[str]
    justificacion: str
    firmas_no_decidibles: FrozenSet[str] = field(default_factory=frozenset)
    conflicto: FrozenSet[str] = field(default_factory=frozenset)


class AuditorLaboratorio:
    """
    Implementación de referencia del AAU v0.1.

    Entrada:
        presv: conjunto de firmas preservables.
        afirmaciones: Activas(G), firmas que el modelo mantiene como vigentes.

    Salida:
        AuditoriaSalida con estado y justificación mínima.
    """

    def __init__(self, presv: Iterable[Invariante], afirmaciones: Iterable[str] | None = None):
        self.presv: Set[Invariante] = set(presv)
        self.activas: Set[str] = set(afirmaciones or [])
        self.grafo_gi: Dict[str, Set[str]] = {}

    def construir_gi(self, inv_satisfechos: Set[Invariante]) -> None:
        """
        Construye G_I como grafo no dirigido de acoplamiento entre invariantes.

        Dos invariantes se conectan si sus embeddings comparten al menos
        un nodo o una arista.
        """
        self.grafo_gi = {f.id: set() for f in inv_satisfechos}
        invariantes = list(inv_satisfechos)

        for i, fa in enumerate(invariantes):
            for fb in invariantes[i + 1:]:
                if self._estan_acoplados(fa, fb):
                    self.grafo_gi[fa.id].add(fb.id)
                    self.grafo_gi[fb.id].add(fa.id)

    @staticmethod
    def _estan_acoplados(fa: Invariante, fb: Invariante) -> bool:
        return bool(fa.embedding_nodes & fb.embedding_nodes) or bool(
            fa.embedding_edges & fb.embedding_edges
        )

    def ejecutar_aau(self) -> AuditoriaSalida:
        """
        Ejecuta el AAU v0.1.

        Jerarquía de estados:
            1. NO AUDITABLE
            2. APROBADO
            3. INCONSISTENTE
            4. SUSPENDIDO
        """
        if not self.presv:
            return AuditoriaSalida(
                estado=ESTADO_NO_AUDITABLE,
                componente_ruptura=frozenset(),
                justificacion="Presv no definido o vacío.",
            )

        no_decidibles = frozenset(f.id for f in self.presv if not f.is_computable)
        if no_decidibles:
            return AuditoriaSalida(
                estado=ESTADO_NO_AUDITABLE,
                componente_ruptura=frozenset(),
                justificacion=f"Firmas preservables no decidibles: {sorted(no_decidibles)}.",
                firmas_no_decidibles=no_decidibles,
            )

        inv_g = {f for f in self.presv if f.is_satisfied}

        if not inv_g:
            return AuditoriaSalida(
                estado=ESTADO_NO_AUDITABLE,
                componente_ruptura=frozenset(),
                justificacion="No hay invariantes satisfechos auditables.",
            )

        self.construir_gi(inv_g)

        rupturas = {f for f in self.presv if not f.is_satisfied}

        if not rupturas:
            return AuditoriaSalida(
                estado=ESTADO_APROBADO,
                componente_ruptura=frozenset(),
                justificacion="Sin rupturas detectadas.",
            )

        break_fk: Set[str] = set()
        for f in rupturas:
            if f.id in self.grafo_gi:
                break_fk.update(self._obtener_componente_conexa(f.id))
            else:
                # Firma rota que no pertenece a Inv(G_new). Se reporta como ruptura local.
                break_fk.add(f.id)

        conflicto = frozenset(self.activas & break_fk)

        if conflicto:
            return AuditoriaSalida(
                estado=ESTADO_INCONSISTENTE,
                componente_ruptura=frozenset(break_fk),
                justificacion=f"Activas(G) intersecta Break(Fk): {sorted(conflicto)}.",
                conflicto=conflicto,
            )

        return AuditoriaSalida(
            estado=ESTADO_SUSPENDIDO,
            componente_ruptura=frozenset(break_fk),
            justificacion="Zona de riesgo estructural detectada sin contradicción declarativa.",
        )

    def _obtener_componente_conexa(self, start_id: str) -> Set[str]:
        visitados: Set[str] = set()
        pila = [start_id]

        while pila:
            actual = pila.pop()

            if actual in visitados:
                continue

            visitados.add(actual)

            for vecino in self.grafo_gi.get(actual, set()):
                if vecino not in visitados:
                    pila.append(vecino)

        return visitados

```

```text
from IMPLEMENTACION.auditor_laboratorio import (
    Invariante,
    AuditorLaboratorio,
    ESTADO_APROBADO,
    ESTADO_SUSPENDIDO,
    ESTADO_INCONSISTENTE,
    ESTADO_NO_AUDITABLE,
)


def inv(id, computable=True, satisfied=True, nodes=(), edges=()):
    return Invariante(
        id=id,
        is_computable=computable,
        is_satisfied=satisfied,
        embedding_nodes=frozenset(nodes),
        embedding_edges=frozenset(edges),
    )


def test_no_auditable_por_presv_vacio():
    salida = AuditorLaboratorio(set(), set()).ejecutar_aau()
    assert salida.estado == ESTADO_NO_AUDITABLE


def test_no_auditable_por_firma_no_decidible():
    f1 = inv("F1", computable=False, nodes=("A",))
    salida = AuditorLaboratorio({f1}, set()).ejecutar_aau()
    assert salida.estado == ESTADO_NO_AUDITABLE
    assert salida.firmas_no_decidibles == frozenset({"F1"})


def test_no_auditable_sin_invariantes_satisfechos():
    f1 = inv("F1", satisfied=False, nodes=("A",))
    salida = AuditorLaboratorio({f1}, set()).ejecutar_aau()
    assert salida.estado == ESTADO_NO_AUDITABLE


def test_aprobado_sin_rupturas():
    f1 = inv("F1", nodes=("A",))
    f2 = inv("F2", nodes=("B",))
    salida = AuditorLaboratorio({f1, f2}, set()).ejecutar_aau()
    assert salida.estado == ESTADO_APROBADO
    assert salida.componente_ruptura == frozenset()


def test_suspendido_por_ruptura_sin_conflicto():
    # F1 roto, F2 satisfecho. No hay Activas(G) que niegue la zona de riesgo.
    f1 = inv("F1", satisfied=False, nodes=("A",))
    f2 = inv("F2", satisfied=True, nodes=("A",))
    salida = AuditorLaboratorio({f1, f2}, set()).ejecutar_aau()
    assert salida.estado == ESTADO_SUSPENDIDO
    assert "F1" in salida.componente_ruptura


def test_inconsistente_por_activa_en_break():
    f1 = inv("F1", satisfied=False, nodes=("A",))
    f2 = inv("F2", satisfied=True, nodes=("A",))
    salida = AuditorLaboratorio({f1, f2}, {"F1"}).ejecutar_aau()
    assert salida.estado == ESTADO_INCONSISTENTE
    assert salida.conflicto == frozenset({"F1"})


def test_gi_acopla_por_nodo_compartido():
    f1 = inv("F1", nodes=("A", "B"))
    f2 = inv("F2", nodes=("B", "C"))
    auditor = AuditorLaboratorio({f1, f2}, set())
    auditor.construir_gi({f1, f2})
    assert auditor.grafo_gi["F1"] == {"F2"}
    assert auditor.grafo_gi["F2"] == {"F1"}


def test_gi_no_acopla_si_no_comparten_embedding():
    f1 = inv("F1", nodes=("A",))
    f2 = inv("F2", nodes=("B",))
    auditor = AuditorLaboratorio({f1, f2}, set())
    auditor.construir_gi({f1, f2})
    assert auditor.grafo_gi["F1"] == set()
    assert auditor.grafo_gi["F2"] == set()
```
